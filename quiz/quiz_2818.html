関数定義には、関数につけられた識別子、処理内容、仮引数リスト、戻り値を記述する。

// C言語

// 関数定義
int add( int a_, int b_ )
{
    int result = a_ + b_;
    return result;
}

<?php

// PHP

// 関数定義
function add( $a_, $b_ )
{
    $result = $a_ + $b_;
    return $result;
}

?>

// JavaScript

function add( a_, b_ )
{
    let result = a_ + b_;
    return result;
}
