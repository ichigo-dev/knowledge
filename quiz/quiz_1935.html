リゾルバは、GraphQLにおいて、Queryを解析して処理を行う関数。GraphQLを扱う実行環境で、外部のプログラムがリゾルバとして扱われる場合が多い。

以下は、TypeScript（Apolloを使用した場合）にてリゾルバを記述する例。

import { ApolloServer } from '@apollo/server';

// GraphQL
const gql = `#graphql
    type Person
    {
        name: String!
        age: Int
    }

    type Query
    {
        persons: [Person]
    }
`;

// Data source
const persons =
[
    {
        "name": "Smith",
        "age": 20,
    },
    {
        "name": "Johnson",
        "age": 40,
    },
];

// Resolver
const resolvers =
{
    Query:
    {
        persons: () => persons,
    },
};

// GraphQL server
const server = new ApolloServer(
{
    gql,
    resolvers,
});
