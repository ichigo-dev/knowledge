テーブルのデータを抽出して取得するには、WHEREコマンドを使用する。

SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [condition];

-- OR条件
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [condition1] OR
    [condition2] OR
    ...;

-- AND条件
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [condition1] AND
    [condition2] AND
    ...;

-- NOT条件
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    NOT [condition];

-- 範囲条件
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [column] BETWEEN [value1] AND [value2];

-- 含有条件
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [column] IN ([value1], [value2], ...);

-- ワイルドカード
-- '%': 0文字以上の任意の文字列
-- '_': 任意の1文字
SELECT
    [column1],
    [column2],
    ...
FROM
    [table]
WHERE
    [column] LIKE '%[value]%';
