テーブルのデータを集約して様々な統計データを取得するには、SUM、AVG、MAX、MIN、COUNTといったコマンドを使用する。

-- 合計値
SELECT
    SUM([column])
FROM
    [table];

-- 平均値
SELECT
    AVG([column])
FROM
    [table];

-- 最大値
SELECT
    MAX([column])
FROM
    [table];

-- 最小値
SELECT
    MIN([column])
FROM
    [table];

-- データ数
SELECT
    COUNT([column])
FROM
    [table];
